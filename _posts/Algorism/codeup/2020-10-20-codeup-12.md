        int num1 = sc.nextInt();
        int num2 = sc.nextInt();
        int sum = num1 + num2;
        
        if(num1 % 2 == 0) System.out.print("짝수+");
        else System.out.print("홀수+");
        
        if(num2 % 2 == 0) System.out.print("짝수=");
        else System.out.print("홀수=");
        
        if(sum % 2 == 0) System.out.print("짝수");
        else System.out.print("홀수");
        
	}
}
