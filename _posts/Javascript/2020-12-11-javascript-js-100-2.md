# 문제 16
```javascript
<script>
    var str = prompt('문장을 입력해주세요.');
    var rStr ='';
    for(var i = str.length -1; i >= 0; i--){
        rStr+=str[i];
    }
    console.log(rStr);
</script>
```

# 문제 18
```javascript
<script>
    var score = prompt('점수를 입력하세요.').split(" ");
    var sum = 0;

    for(var i = 0; i < score.length; i++){
        sum += Number(score[i]);
    }
    console.log(parseInt(sum/3));
</script>
```
